        \item $(\forall x (x = y)) \rightarrow (x \in y)$

        \item $(x \in y)$
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item $\forall x ((x = y) \rightarrow (x \in y))$
        \item $(x = y) \rightarrow (x \in y)$
        \item $(x = y)$
        \item $(x \in y)$
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item $\forall x (x = x) \rightarrow (\forall y \exists z (((x = y) \land (y = z)) \rightarrow \neg (x \in y)))$
        \item $\forall x (x = x)$
        \item $(x = y)$
        \item $\forall z \exists y (((x = y) \land (y = z)) \rightarrow \neg (x \in y))$
        \item $((x = y) \land (y = z))$
        \item $(x = y)$
        \item $(y = z)$
        \item $\neg (x \in y)$
        \item $(x \in y)$
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item $(x = y) \rightarrow \exists y (x = y)$
        \item $(x = y)$
        \item $\exists y (x = y)$
        \item $(x = y)$
    \end{enumerate}
\end{enumerate}
